42 template<typename RNG>
43 Path &PathGenerator<RNG>::next() {
44  if (!isAntithetic_) {
45  vector<vector<Quote> >::iterator it_value = next_->getValues().begin();
46  vector<Quote> x0 = model_->getInitial();
47  *it_value = x0;
48  for (vector<Time>::const_iterator it_time = next_->getTimeGrid().begin();
49  it_time != next_->getTimeGrid().end() - 1; it_time++) {
50  it_value++;
51  Time dt = *(it_time + 1) - *it_time;
52  double dw = rng_.next() * sqrt(dt);
53  *it_value = model_->evolve(*it_time, x0, dt, dw);
54  x0 = *it_value;
55  }
56  }
57  else {
58  vector<vector<Quote> >::iterator it_value = next_->getValues().begin();
59  vector<vector<Quote> >::iterator it_anti_value = next_->getAntitheticValues().begin();
60  vector<Quote> x0_val = model_->getInitial();
61  vector<Quote> x0_antiVal = x0_val;
62  *it_value = x0_val;
63  *it_anti_value = x0_antiVal;
64  for (vector<Time>::const_iterator it_time = next_->getTimeGrid().begin();
65  it_time != next_->getTimeGrid().end() - 1; it_time++) {
66  it_value++;
67  it_anti_value++;
68  Time dt = *(it_time + 1) - *it_time;
69  double dw = rng_.next() * sqrt(dt);
70  *it_value = model_->evolve(*it_time, x0_val, dt, dw);
71  *it_anti_value = model_->evolve(*it_time, x0_antiVal, dt, -dw);
72  x0_val = *it_value;
73  x0_antiVal = *it_anti_value;
74  }
75  }
76  return *next_;
77 }
